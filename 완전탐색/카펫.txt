#include <string>
#include <vector>

using namespace std;

vector<int> solution(int brown, int yellow) {
    vector<int> answer;
    for (int i = 1; i * i <= yellow; i++) {
        if (yellow % i == 0) {
            int y_width = max(i, yellow / i);
            int y_height = min(i, yellow / i);
            int b_width = y_width + 2;
            int b_height = y_height;
            if ((b_width + b_height)*2 == brown) {
                answer.push_back(b_width);
                answer.push_back(b_height+2);
            }
;        }
    }
    return answer;
}
