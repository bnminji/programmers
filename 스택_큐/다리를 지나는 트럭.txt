#include <string>
#include <vector>
#include <queue>
using namespace std;

int solution(int bridge_length, int weight, vector<int> truck_weights) {
    int answer = 0;
    queue<int> q;
    queue<int> trucks;

    for (int i = 0; i < truck_weights.size(); i++) {
        trucks.push(truck_weights[i]);
    }

    int num = 0; int cnt = 0; 
    while (!trucks.empty()) {
        if (num + trucks.front() <= weight) {
            q.push(trucks.front());
            num += trucks.front(); trucks.pop();
        }
        else {
            q.push(-1);
        } 
        cnt++;
        if (cnt < bridge_length) {
            continue;
        }
        if(q.front() != -1) num -= q.front();
        q.pop();
    }
    cnt += bridge_length;
    answer = cnt;
    return answer;
}
