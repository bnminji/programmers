#include <string>
#include <vector>
#include <math.h>
#include <queue>
using namespace std;

vector<int> solution(vector<int> progresses, vector<int> speeds) {
    vector<int> answer;
    queue<int> q;
    for (int i = 0; i < progresses.size(); i++) {
        int day = ceil((100 - (double)progresses[i]) / speeds[i]);
        q.push(day);
    }

    while(!q.empty()){
        int cnt = 1;
        int day = q.front(); q.pop();
        while (!q.empty() && day >= q.front()) { q.pop(); cnt++; }
        answer.push_back(cnt);
    }
    
    return answer;
}
